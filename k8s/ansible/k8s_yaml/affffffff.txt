apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: eschool-ingress
  namespace: eschool
  annotations:
    kubernetes.io/ingress.class: nginx
    # для Angular/React SPA — всі непізнані шляхи йдуть на index.html
    nginx.ingress.kubernetes.io/rewrite-target: /$1
    nginx.ingress.kubernetes.io/use-regex: "true"
spec:
  rules:
    - http:
        paths:
          # API бекенд
          - path: /api/(.*)
            pathType: ImplementationSpecific
            backend:
              service:
                name: backend
                port:
                  number: 8080

          # Всі інші шляхи — фронтенд
          - path: /(.*)
            pathType: ImplementationSpecific
            backend:
              service:
                name: frontend
                port:
                  number: 80
